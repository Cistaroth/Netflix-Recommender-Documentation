TMDB_REQUEST_URL_ID = (
    "https://api.themoviedb.org/3/{media_type}/{show_id}?append_to_response=credits"
)
TMDB_REQUEST_URL_TITLE = "https://api.themoviedb.org/3/search/multi?query={title}"
TMDB_REQUEST_URL_GENRE_MAP = "https://api.themoviedb.org/3/genre/{media_type}/list"
TMDB_REQUEST_URL_FETCH_NEW = "https://api.themoviedb.org/3/discover/{media_type}?page={page}&sort_by=popularity.desc"

def construct_genre_map(api_key: str) -> dict[str, dict[int, str]]:
    """
    Construct a dictionary mapping genre IDs to names for movies and TV shows

    Args:
        api_key (str): API key for The Movie Database

    Returns:
        dict[str, dict[int, str]]: Dictionary mapping genre IDs to names for movies and TV shows
    """
    genre_map = {}

    # Construct genre map for movies and TV shows
    for media_type in ["movie", "tv"]:
        # Fetch genre map for media type
        request_url = TMDB_REQUEST_URL_GENRE_MAP.format(media_type=media_type)
        response = requests.get(request_url, params={"api_key": api_key})

        # Extract genre map for media type
        response = {genre["id"]: genre["name"] for genre in response.json()["genres"]}

        # Store genre map for media type
        genre_map[media_type] = response

    return genre_map

def TMDB_info_by_ID(
    show_id: int,
    media_type: Literal["tv", "movie"],
    api_key: str,
    cast_limit: int = 5,
) -> dict[str, str | float | int | list | None] | None:
    """
    Fetch information about a movie or TV show from The Movie Database by its ID

    Args:
        show_id (int): ID of the movie or TV show
        media_type (Literal["tv", "movie"]): Type of the movie or TV show
        api_key (str): API key for The Movie Database
        cast_limit (int, optional): Maximum number of people to include in the credits. Defaults to 5.

    Returns:
        dict[str, str | float | int | list | None] | None: Information about the movie or TV show, or None if the request fails
    """
    
    # Fetch information about the movie or TV show
    request_url = TMDB_REQUEST_URL_ID.format(media_type=media_type, show_id=show_id)
    response = requests.get(request_url, params={"api_key": api_key})

    # Extract information about the movie or TV show
    if response.status_code == 200 and response.json():
        item = response.json()

        # Extract information about the movie or TV show
        title = item.get("title", None) or item.get("name", None)
        overview = item.get("overview", None)
        date = item.get("first_air_date", None) or item.get("release_date", None)
        year = int(date[:4]) if date else None
        language = item.get("original_language", None)
        vote_average = item.get("vote_average", None)

        production_countries = [
            production_country["name"]
            for production_country in item.get("production_countries", [])
        ]
        genres = [genre["name"] for genre in item.get("genres", [])]

        credits_dict = item.get("credits", {"cast": [], "crew": []})
        credits_cast = credits_dict["cast"][: cast_limit + 1]
        credits_crew = credits_dict["crew"][: cast_limit + 1]

        credits = set(person["name"] for person in credits_cast + credits_crew)
        credits = list(credits)

        try:
            runtime = int(item.get("runtime", 0)) or item.get("episode_run_time", [])[0]
        except Exception:
            runtime = None

        # Return information about the movie or TV show
        return {
            "title": title,
            "overview": overview,
            "year": year,
            "language": language,
            "media_type": media_type,
            "vote_average": vote_average,
            "production_countries": production_countries,
            "genres": genres,
            "credits": credits,
            "runtime": runtime,
        }

def TMDB_info_by_title(
    title: str, api_key: str, cast_limit: int = 5
) -> dict[str, str | float | int | list | None] | None: 
    """
    Fetch information about a movie or TV show from The Movie Database by its title

    Args:
        title (str): Title of the movie or TV show
        api_key (str): API key for The Movie Database
        cast_limit (int, optional): Maximum number of people to include in the credits. Defaults to 5.

    Returns:
        dict[str, str | float | int | list | None] | None: Information about the movie or TV show, or None if the request fails
    """
    media_type, show_id = None, None

    # Fetch information about the movie or TV show
    request_url = TMDB_REQUEST_URL_TITLE.format(title=title.replace(" ", "+"))
    response = requests.get(request_url, params={"api_key": api_key})

    # Pick the first movie or TV show with an overview
    if response.status_code == 200 and response.json()["results"]:
        results = response.json()["results"]

        for item in results:
            item_media_type = item.get("media_type", None)
            if (
                item_media_type == "tv"
                or item_media_type == "movie"
                and item.get("overview", None)
            ):
                media_type = item_media_type
                show_id = item.get("id", None)

    # If no movie or TV show with an overview is found, return None
    if media_type is None or show_id is None:
        return None

    # Fetch more information about the movie or TV show
    return TMDB_info_by_ID(
        show_id=show_id, media_type=media_type, api_key=api_key, cast_limit=cast_limit
    )

def TMDB_fetch_new_movies(
    api_key: str,
    movie_number: int = 50,
    tv_number: int = 50,
    cast_limit: int = 5,
    verbose: bool = True,
) -> pd.DataFrame:
    """
    Fetch information about new movies or TV shows from The Movie Database

    Args:
        api_key (str): API key for The Movie Database
        movie_number (int, optional): Maximum number of movies to fetch. Defaults to 50.
        tv_number (int, optional): Maximum number of TV shows to fetch. Defaults to 50.
        cast_limit (int, optional): Maximum number of people to include in the credits. Defaults to 5.
        verbose (bool, optional): Whether to print logs to the console. Defaults to True.

    Returns:
        pd.DataFrame: Information about the movies or TV shows, or an empty DataFrame if the request fails
    """
    if verbose:
        log.section(" --- FETCHING NEW SHOWS --- ")

    data = []

    # Fetch information about new movies and TV shows
    for media_type, limit in zip(["movie", "tv"], [movie_number, tv_number]):
        if verbose:
            log.plain(f"FETCHING NEW SHOW OF TYPE {media_type}...")

        # Create a progress bar if verbose
        with tqdm(total=limit, disable=not verbose) as pbar:
            page = 1
            # Fetch information until the limit is reached
            while len(data) < movie_number + tv_number:

                # Fetch information about new movies or TV shows
                request_url = TMDB_REQUEST_URL_FETCH_NEW.format(
                    media_type=media_type, page=page
                )
                response = requests.get(request_url, params={"api_key": api_key})

                if response.status_code == 200 and response.json()["results"]:
                    results = response.json()["results"]

                    for result in results:

                        # Ignore results that are not movies or TV shows
                        if media_type not in ("movie", "tv"):
                            continue
                            
                        # Fetch more information about the movie or TV show
                        show_info = TMDB_info_by_ID(
                            show_id=result["id"],
                            media_type=media_type,
                            api_key=api_key,
                            cast_limit=cast_limit,
                        )

                        # Add information about the movie or TV show to the list
                        if show_info:
                            data.append(show_info)
                            pbar.update(1)
                else:
                    # Break the loop if the request fails
                    break
                
                # Increment the page
                page += 1

    # Return information about the movies or TV shows
    return pd.DataFrame(data)
