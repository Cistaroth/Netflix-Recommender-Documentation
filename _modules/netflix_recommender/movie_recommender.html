    A class for training and evaluating a movie recommendation model.

        plots_save_path (Path): The path to save the plots to.

    def __init__(
        self,

        # Preprocessing models
        self._standard_scaler: StandardScaler = standard_scaler
        self._pca: PCA | None = None

        # Model
        self._model: KNeighborsClassifier | None = None

    def generate_embeddings(
        self, data: pd.DataFrame, columns: list[str], verbose: bool = True
    ) -> dict[str, pd.DataFrame]:
        """
        Generates text embeddings for the specified columns in the provided data.

        Args:
            data (pd.DataFrame): The data to generate embeddings for.
            columns (list[str]): The columns to generate embeddings for.
            verbose (bool, optional): Whether to show progress bars. Defaults to True.

        Returns:
            dict[str, pd.DataFrame]: A dictionary mapping column names to embedded columns. 
                The embedded columns are DataFrames with the same index as the input data, 
                containing the embedded values for each column.
        """
        if verbose:
            log.section(" --- EMBEDDING COLUMNS --- ")

        embeddings = {}
        # Create embeddings for each column
        for column in columns:
            if verbose:
                log.plain(f"Creating embeddings for: {column}")

            # Convert column to list
            column_to_encode = data[column].to_list()

            # Create embeddings and store in DataFrame
            embedded_column = self._embedding_model.encode(
                column_to_encode, show_progress_bar=verbose
            )
            embedded_column = pd.DataFrame(embedded_column)

            # Set column names
            embedded_column.columns = [
                f"{column}_emb_dim_{i}" for i in range(embedded_column.shape[1])
            ]

            embeddings[column] = embedded_column

        return embeddings

    def one_hot_encode_columns(
        self,
        data: pd.DataFrame,
        columns: list[str] = [],
        list_columns: list[str] = [],
        verbose: bool = True,
    ) -> tuple[dict[str, pd.DataFrame], dict[str, pd.DataFrame]]:
        """
        One hot encode columns in a DataFrame.

        Args:
            data (pd.DataFrame): DataFrame to one hot encode columns from.
            columns (list[str], optional): Columns to one hot encode. Defaults to [].
            list_columns (list[str], optional): Columns to one hot encode as lists. Defaults to [].
            verbose (bool, optional): Whether to print logs to the console. Defaults to True.

        Returns:
            tuple[dict[str, pd.DataFrame], dict[str, pd.DataFrame]]: A tuple containing two dictionaries. 
                The first dictionary maps column names to one hot encoded DataFrames. 
                The second dictionary maps column names to one hot encoded DataFrames as lists.
        """
        if verbose:
            log.section(" --- ONE HOT ENCODE COLUMNS --- ")

        ohe_columns, ohe_list_columns = {}, {}

        if verbose and columns:
            log.plain("ONE HOT ENCODING COLUMNS... ")
        
        # One hot encode columns
        for column in columns:
            # Convert column to string
            data[column] = data[column].astype(str)

            # Check if column has already been one hot encoded
            if column not in self._one_hot_encoders:
                # Create one hot encoder
                self._one_hot_encoders[column] = OneHotEncoder(
                    sparse_output=False, handle_unknown="ignore"
                )

                # One hot encode column
                column_encoded = self._one_hot_encoders[column].fit_transform(
                    data[[column]]
                )
            else:
                # One hot encode column
                column_encoded = self._one_hot_encoders[column].transform(
                    data[[column]]
                )

            # Create one hot encoded DataFrame
            one_hot_encoder = self._one_hot_encoders[column]
            column_encoded_df = pd.DataFrame(
                np.array(column_encoded),
                columns=one_hot_encoder.get_feature_names_out([column]),
            )

            # Add one hot encoded DataFrame to dictionary
            ohe_columns[column] = column_encoded_df

        if verbose and list_columns:
            log.plain("ONE HOT ENCODING LIST COLUMNS... ")

        # One hot encode list columns
        for column in list_columns:
            # Convert column to string
            data[column] = data[column].astype(str)

            # Check if column has already been one hot encoded
            if column not in self._one_hot_list_encoders:
                # Create one hot encoder
                self._one_hot_list_encoders[column] = MultiLabelBinarizer()

                # One hot encode column
                column_encoded = self._one_hot_list_encoders[column].fit_transform(
                    data[column]
                )
            else:
                # One hot encode column
                column_encoded = self._one_hot_list_encoders[column].transform(
                    data[column]
                )

            # Create one hot encoded DataFrame
            multi_label_binarizer = self._one_hot_list_encoders[column]
            column_encoded_df = pd.DataFrame(
                np.array(column_encoded), columns=multi_label_binarizer.classes_
            )

            # Add one hot encoded DataFrame to dictionary
            ohe_list_columns[column] = column_encoded_df

        return ohe_columns, ohe_list_columns



    def _validate_train_data(self, train_x: pd.DataFrame, train_y: pd.Series) -> None:
        """
        Validates the train data by checking if the lengths of train_x and train_y match, 
        and if the values of train_y are either 0 or 1.

        Args:
            train_x (pd.DataFrame): The training data.
            train_y (pd.Series): The training labels.

        Raises:
            InvalidTrainDataError: If the lengths of train_x and train_y do not match, 
                or if the values of train_y are not either 0 or 1.
        """
        if len(train_x) != len(train_y):
            raise InvalidTrainDataError(message=TRAIN_DATA_NON_MATCHING_LENGTH_MSG)
        
        if set(train_y.values) != set([0, 1]):
            raise InvalidTrainDataError(message=TRAIN_Y_INVALID_VALUES_MSG)

    def train(
        self,
        train_x: pd.DataFrame,
        train_y: pd.Series,
        cross_validation_grid: CrossValidationGrid = CrossValidationGrid(),
        plot: bool = True,
        verbose: bool = True,
    ) -> None:
        """
        Trains a classification model on the given data.

        Args:
            train_x (pd.DataFrame): The training data.
            train_y (pd.Series): The training labels.
            cross_validation_grid (CrossValidationGrid, optional): The cross validation grid to use. 
                Defaults to CrossValidationGrid().
            plot (bool, optional): Whether to plot the cross validation results. Defaults to True.
            verbose (bool, optional): Whether to print logs to the console. Defaults to True.

        Raises:
            InvalidTrainDataError: If the lengths of train_x and train_y do not match, 
                or if the values of train_y are not either 0 or 1.
        """
        if verbose:
            log.section(" --- Training Classification Model --- ")

        # Validate train data
        self._validate_train_data(train_x=train_x, train_y=train_y)

        # Standardize train data
        train_x_standard_scaled = self._standard_scaler.fit_transform(X=train_x)

        # Cross validate and select optimal model
        self._cross_validate_and_select(
            train_x_standard_scaled=train_x_standard_scaled,
            train_y=train_y,
            cross_validation_grid=cross_validation_grid,
            plot=plot,
            verbose=verbose,
        )



    def _cross_validate_and_select(
        self,
        train_x_standard_scaled: np.ndarray,
        train_y: pd.Series,
        cross_validation_grid: CrossValidationGrid,
        plot: bool = True,
        verbose: bool = True,
    ) -> None:
        """
        Cross validates the model and selects the optimal hyperparameters.

        Args:
            train_x_standard_scaled (np.ndarray): The preprocessed training data.
            train_y (pd.Series): The training labels.
            cross_validation_grid (CrossValidationGrid): The cross validation grid to use.
            plot (bool, optional): Whether to plot the cross validation results. Defaults to True.
            verbose (bool, optional): Whether to print logs to the console. Defaults to True.

        Returns:
            None
        """

        if verbose:
            log.section(" --- CROSS VALIDATING AND SELECTING OPTIMAL MODEL --- ")
            log.plain("Cross validating to find optimal parameters...")
            hyperparameter_combinations = tqdm(cross_validation_grid.product)
        else:
            hyperparameter_combinations = cross_validation_grid.product

        results = []

        # Cross validate by fitting a model for each combination of hyperparameters
        for properties in hyperparameter_combinations:
            # Unpack hyperparameters
            knn_n_neighbours, knn_weight, knn_metric, pca_n_components, *_ = properties

            # PCA preprocessing
            pca = PCA(n_components=pca_n_components)
            train_x_preprocessed = pca.fit_transform(train_x_standard_scaled, train_y)

            # Initialize model
            model = KNeighborsClassifier(
                n_neighbors=knn_n_neighbours, weights=knn_weight, metric=knn_metric
            )

            # Cross validate
            scores = cross_val_score(
                estimator=model,
                X=train_x_preprocessed,
                y=train_y,
                cv=cross_validation_grid.folds,
                scoring=cross_validation_grid.validation_metric,
            )
            mean_score = np.mean(scores)

            # Store results
            results.append(
                {
                    "pca_n_components": pca_n_components,
                    "knn_n_neighbors": knn_n_neighbours,
                    "knn_weights": knn_weight,
                    "knn_metrics": knn_metric,
                    "cv_scores": mean_score,
                }
            )

        # Convert results to DataFrame
        results_df = pd.DataFrame(results)

        # Plot cross validation results
        if plot:
            if verbose:
                log.section(" --- PLOTTING RESULT OF CROSS VALIDATION --- ")

            self._validation_plots(
                results=results_df, scoring=cross_validation_grid.validation_metric
            )

        if verbose:
            log.section(" --- CHOOSING OPTIMAL MODEL --- ")

        # Select optimal model
        if cross_validation_grid._best_row_strategy == "min":
            best_row = results_df.loc[results_df["cv_scores"].idxmin()]
        else:
            best_row = results_df.loc[results_df["cv_scores"].idxmax()]

        # Unpack optimal hyperparameters
        best_n_components = best_row["pca_n_components"].item()
        best_n_neighbors = best_row["knn_n_neighbors"].item()
        best_weight = cast(Literal["uniform", "distance"], best_row["knn_weights"])
        best_metric = str(best_row["knn_metrics"])

        # Store optimal model
        self._pca = PCA(n_components=best_n_components)
        self._model = KNeighborsClassifier(
            n_neighbors=best_n_neighbors, weights=best_weight, metric=best_metric
        )

    def _validation_plots(self, results: pd.DataFrame, scoring: str) -> None:
        """
        Plots the results of the cross validation.

        The results are plotted in a grid where the x-axis is the hyperparameter being varied,
        the y-axis is the scoring metric, and each subplot is a different combination of hyperparameters.

        The top row of subplots shows the scoring metric vs the number of nearest neighbors for KNN,
        and the bottom row shows the scoring metric vs the number of components for PCA.

        Args:
            results (pd.DataFrame): The DataFrame containing the results of the cross validation.
            scoring (str): The scoring metric used for cross validation.

        Returns:
            None
        """

        # Get unique values of hyperparameters        
        knn_weights_unique = sorted(results["knn_weights"].unique())
        knn_metrics_unique = sorted(results["knn_metrics"].unique())

        # Create subplots
        nrows = len(knn_metrics_unique)
        ncols = len(knn_weights_unique) * 2

        fig, axes = plt.subplots(
            nrows=nrows, ncols=ncols, figsize=(3 * ncols, 4 * nrows), sharey=True
        )

        if nrows == 1:
            axes = axes.reshape(-1, 1)

        # Plot results
        for row, metric in enumerate(knn_metrics_unique):
            for column, weight in enumerate(knn_weights_unique):
                subset = results[
                    (results["knn_metrics"] == metric)
                    & (results["knn_weights"] == weight)
                ]

                # Scoring vs K
                k_scores = (
                    subset.groupby("knn_n_neighbors")["cv_scores"].mean().reset_index()
                )

                axes_k = axes[row, column * 2]

                axes_k.plot(
                    k_scores["knn_n_neighbors"], k_scores["cv_scores"], marker="o"
                )

                axes_k.set_title(f"{metric} | weight={weight} \n {scoring} vs K")
                axes_k.set_xlabel("Number of Neighbors")
                if column == 0:
                    axes_k.set_ylabel(f"CV {scoring}")
                axes_k.grid(True)

                # Scoring vs PCA
                pca_scores = (
                    subset.groupby("pca_n_components")["cv_scores"].mean().reset_index()
                )

                axes_pca = axes[row, column * 2 + 1]
                axes_pca.plot(
                    pca_scores["pca_n_components"], pca_scores["cv_scores"], marker="o"
                )

                axes_pca.set_title(f"{metric} | weight={weight} \n {scoring} vs PCA")
                axes_pca.set_xlabel("PCA n_components")
                axes_pca.grid(True)

        # Adjust spacing and save
        plt.tight_layout()
        plt.savefig(self.plots_save_path / "cross_validation_sumary.png")

    def predict(self, predict_x: pd.DataFrame, verbose: bool = True) -> np.ndarray:
        """
        Predicts the ratings for a given set of movies.

        Args:
            predict_x (pd.DataFrame): The DataFrame containing the movies to predict ratings for.
            verbose (bool, optional): Whether or not to print the steps of the prediction process. Defaults to True.

        Returns:
            np.ndarray: The predicted ratings.

        Raises:
            ModelNotFitError: If the model has not been fit before calling this method.
        """
        if verbose:
            log.section(" --- PREDICTING --- ")
        if self._standard_scaler is None or self._pca is None or self._model is None:
            raise ModelNotFitError(MODEL_NOT_FITTED_MSG)

        predict_x_standard_scaled = self._standard_scaler.transform(predict_x)
        predict_x_preprocessed = self._pca.transform(predict_x_standard_scaled)
        return self._model.predict(predict_x_preprocessed)

    def preprocess_train_predict(
        self,
        train_x: pd.DataFrame,
        train_y: pd.Series,
        predict_x: pd.DataFrame,
        embedding_columns: list[str] = [],
        one_hot_encode_columns: list[str] = [],
        one_hot_encode_list_columns: list[str] = [],
        cross_validation_grid: CrossValidationGrid = CrossValidationGrid(),
        plot: bool = True,
        verbose: bool = True,
    ) -> np.ndarray:
        """
        Classifies matches based on the given training data and predicts ratings for a given set of movies.

        Args:
            train_x (pd.DataFrame): The DataFrame containing the training data.
            train_y (pd.Series): The Series containing the training labels.
            predict_x (pd.DataFrame): The DataFrame containing the movies to predict ratings for.
            embedding_columns (list[str], optional): The columns to generate embeddings for. Defaults to [].
            one_hot_encode_columns (list[str], optional): The columns to one-hot encode. Defaults to [].
            one_hot_encode_list_columns (list[str], optional): The list columns to one-hot encode. Defaults to [].
            cross_validation_grid (CrossValidationGrid, optional): The cross validation grid to use. Defaults to CrossValidationGrid().
            plot (bool, optional): Whether to plot the cross validation results. Defaults to True.
            verbose (bool, optional): Whether to print logs to the console. Defaults to True.

        Returns:
            np.ndarray: The predicted ratings.
        """

        if verbose:
            log.section(" --- CLASSIFYING MATCHES --- ")

        # Preprocess
        data = {"train_x": train_x.dropna(), "predict_x": predict_x.dropna()}
        for data_type in data:
            if verbose:
                log.section(f" --- PREPROCESSING {data_type} --- ")

            # Embeddings
            if embedding_columns:

                # Generate embeddings
                embeddings_map = self.generate_embeddings(
                    data=data[data_type], columns=embedding_columns, verbose=verbose
                )

                # Drop original columns
                data[data_type] = data[data_type].drop(embedding_columns, axis=1)

                # Add embeddings
                data[data_type] = pd.concat(
                    [data[data_type], *embeddings_map.values()], axis=1
                )

            # One-hot encoding
            if one_hot_encode_columns or one_hot_encode_list_columns:
                # One-hot encode
                one_hot_encoded_columns, one_hot_encoded_list_columns = (
                    self.one_hot_encode_columns(
                        data=data[data_type],
                        columns=one_hot_encode_columns,
                        list_columns=one_hot_encode_list_columns,
                        verbose=verbose,
                    )
                )

                # Drop original columns
                data_df = data[data_type].drop(
                    one_hot_encode_columns + one_hot_encode_list_columns, axis=1
                )

                # Add one-hot encoded columns
                data[data_type] = pd.concat(
                    [data_df]
                    + [*one_hot_encoded_columns.values()]
                    + [*one_hot_encoded_list_columns.values()],
                    axis=1,
                )

        # Split train and predict
        train_x, predict_x = data["train_x"], data["predict_x"]

        # Remove rows with missing values in train
        training_data = pd.concat([train_x, train_y], axis=1).dropna()
        train_x, train_y = training_data.iloc[:, :-1], training_data.iloc[:, -1]

        # Remove rows with missing values in predict
        predict_x = predict_x.dropna()

        # Train
        self.train(
            train_x=train_x,
            train_y=train_y,
            cross_validation_grid=cross_validation_grid,
            plot=plot,
            verbose=verbose,
        )

        # Predict
        return self.predict(predict_x=predict_x, verbose=verbose)

    def preprocess_train_suggest(
        self,
        train_x: pd.DataFrame,
        train_y: pd.Series,
        predict_x: pd.DataFrame,
        embedding_columns: list[str] = [],
        one_hot_encode_columns: list[str] = [],

        cross_validation_grid: CrossValidationGrid = CrossValidationGrid(),
        plot: bool = True,
        verbose: bool = True,
    ) -> pd.DataFrame:
        """
        Preprocesses the data for training the model and making predictions.

        Args:
            train_x (pd.DataFrame): The training data.
            train_y (pd.Series): The labels for the training data.
            predict_x (pd.DataFrame): The data to make predictions on.
            embedding_columns (list[str], optional): The columns to generate embeddings for. Defaults to [].
            one_hot_encode_columns (list[str], optional): The columns to one-hot encode. Defaults to [].
            one_hot_encode_list_columns (list[str], optional): The columns to one-hot encode as lists. Defaults to [].
            cross_validation_grid (CrossValidationGrid, optional): The cross-validation grid to use. Defaults to CrossValidationGrid().
            plot (bool, optional): Whether to plot the results of cross-validation. Defaults to True.
            verbose (bool, optional): Whether to show progress bars. Defaults to True.

        Returns:
            pd.DataFrame: The data with the predicted labels, filtered to only include the rows with a positive prediction.
        """

        # Obtain predictions
        y_pred = self.preprocess_train_predict(
            train_x=train_x,
            train_y=train_y,
            predict_x=predict_x,
            embedding_columns=embedding_columns,
            one_hot_encode_columns=one_hot_encode_columns,
            one_hot_encode_list_columns=one_hot_encode_list_columns,
            cross_validation_grid=cross_validation_grid,
            plot=plot,
            verbose=verbose,
        )

        # Filter
        selected = predict_x[y_pred == 1]
        return selected

    def obtain_classfication_report(
        self,
        y_true: np.ndarray,
        y_pred: np.ndarray,
        plot: bool = True,
        verbose: bool = True,
    ) -> np.ndarray:
        """
        Obtains a classification report for the given true and predicted labels.

        Args:
            y_true (np.ndarray): The true labels.
            y_pred (np.ndarray): The predicted labels.
            plot (bool, optional): Whether to plot the confusion matrix. Defaults to True.
            verbose (bool, optional): Whether to show the classification report. Defaults to True.

        Returns:
            np.ndarray: The confusion matrix.
        """

        # Confusion matrix
        confusion_matrix_array = confusion_matrix(y_true=y_true, y_pred=y_pred)

        # Plot
        if plot:
            plt.clf()
            confusion_matrix_display = ConfusionMatrixDisplay(
                confusion_matrix=confusion_matrix_array, display_labels=[True, False]
            )
            confusion_matrix_display.plot()
            plt.savefig(self.plots_save_path / "confusion_matrix.png")

        # Classification report
        if verbose:
            log.section(" --- CLASSIFICATION REPORT --- ")
            print(classification_report(y_true=y_true, y_pred=y_pred))

        # Return confusion matrix
        return confusion_matrix_array

    def train_evaluate(
        self,
        data_x: pd.DataFrame,
        data_y: pd.Series,
        embedding_columns: list[str] = [],
        one_hot_encode_columns: list[str] = [],
        one_hot_encode_list_columns: list[str] = [],
        cross_validation_grid: CrossValidationGrid = CrossValidationGrid(),
        test_size: float = 0.2,
        random_state: int = 42,
        plot: bool = True,
        verbose: bool = True,
    ) -> np.ndarray:
        
        """
        Trains a model and evaluates its performance on a test set.

        Args:
            data_x (pd.DataFrame): The feature data.
            data_y (pd.Series): The target data.
            embedding_columns (list[str], optional): The columns to generate embeddings for. Defaults to [].
            one_hot_encode_columns (list[str], optional): The columns to one-hot encode. Defaults to [].
            one_hot_encode_list_columns (list[str], optional): The list columns to one-hot encode. Defaults to [].
            cross_validation_grid (CrossValidationGrid, optional): The cross validation grid to use. Defaults to CrossValidationGrid().
            test_size (float, optional): The proportion of the data to include in the test set. Defaults to 0.2.
            random_state (int, optional): The random state to use when splitting the data. Defaults to 42.
            plot (bool, optional): Whether to plot the confusion matrix. Defaults to True.
            verbose (bool, optional): Whether to print logs to the console. Defaults to True.

        Returns:
            np.ndarray: The confusion matrix.
        """

        # Train-test split
        train_x, test_x, train_y, test_y = train_test_split(
            data_x, data_y, random_state=random_state, test_size=test_size, shuffle=True
        )

        # Preprocess, train and predict
        y_pred = self.preprocess_train_predict(
            train_x=train_x,
            train_y=train_y,
            predict_x=test_x,
            embedding_columns=embedding_columns,
            one_hot_encode_columns=one_hot_encode_columns,
            one_hot_encode_list_columns=one_hot_encode_list_columns,
            cross_validation_grid=cross_validation_grid,
            plot=plot,
            verbose=verbose,
        )

        # Obtain classification report
        return self.obtain_classfication_report(
            y_true=test_y, y_pred=y_pred, plot=plot, verbose=verbose
        )
