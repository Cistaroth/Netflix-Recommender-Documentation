class CrossValidationGrid:
    """
    A class to represent a grid of hyperparameters for cross-validation.

    Attributes:
        folds (int): The number of folds for cross-validation.
        validation_metric (str): The metric to use for validation.
        best_row_strategy (Literal["max", "min"]): The strategy to use for selecting the best row.
        knn_n_neighbours (list[int] | range): The number of nearest neighbours for KNN.
        knn_weights (list[Literal["uniform", "distance"]]): The weights for KNN.
        knn_metrics (list[str]): The metrics for KNN.
        pca_n_components (list[float | int] | range): The number of components for PCA.
    """
    def __init__(
        self,
        folds: int = 5,
        validation_metric: str = "f1",
        best_row: Literal["max", "min"] = "max",
        knn_n_neighbours: list[int] | range = [5],
        knn_weights: list[Literal["uniform", "distance"]] = ["distance"],
        knn_metrics: list[str] = ["minkowski"],
        pca_n_components: list[float | int] | range = [0.95],
    ) -> None:
        """
        Initializes the CrossValidationGrid with the given hyperparameters.

        Args:
            folds (int, optional): The number of folds for cross-validation. Defaults to 5.
            validation_metric (str, optional): The metric to use for validation. Defaults to "f1".
            best_row_strategy (Literal["max", "min"], optional): The strategy to use for selecting the best row. Defaults to "max".
            knn_n_neighbours (list[int] | range, optional): The number of nearest neighbours for KNN. Defaults to [5].
            knn_weights (list[Literal["uniform", "distance"]], optional): The weights for KNN. Defaults to ["distance"].
            knn_metrics (list[str], optional): The metrics for KNN. Defaults to ["minkowski"].
            pca_n_components (list[float | int] | range, optional): The number of components for PCA. Defaults to [0.95].
        """
        self._folds = folds
        self._validation_metric = validation_metric
        self._best_row_strategy = best_row
        self._knn_n_neighbours = knn_n_neighbours
        self._knn_weights = knn_weights
        self._knn_metrics = knn_metrics
        self._pca_n_components = pca_n_components

    @property
    def product(self) -> list[tuple]:
        """
        Returns the product of the hyperparameters.

        This is a list of tuples, where each tuple contains one value from each of the hyperparameters.
        The length of the returned list is the product of the lengths of the hyperparameters.

        Returns:
            list[tuple]: The product of the hyperparameters.
        """
        properties = [
            self._knn_n_neighbours,
            self._knn_weights,
            self._knn_metrics,
            self._pca_n_components,
        ]
        return list(product(*properties))

    @property
    def folds(self) -> int:
        """
        The number of folds for cross-validation.

        Returns:
            int: The number of folds.
        """
        return self._folds

    @property
    def validation_metric(self) -> str:
        """
        The validation metric for cross-validation.

        This is the metric used to evaluate the model during cross-validation.
        The possible values are the same as those for the 'scoring' parameter in
        sklearn.model_selection.cross_val_score.

        Returns:
            str: The validation metric.
        """
        return self._validation_metric

    @property
    def best_row_strategy(self) -> str:
        """
        The strategy to use for selecting the best row.

        The possible values are 'max' and 'min'. 'max' will select the row with the highest score,
        while 'min' will select the row with the lowest score.

        Returns:
            str: The strategy to use for selecting the best row.
        """
        return self._best_row_strategy

    @property
    def knn_n_neighbours(self) -> list[int] | range:
        
        """
        The number of nearest neighbours to consider when using the K-Nearest Neighbors algorithm.

        This can be either a list of integers or a range object. If a list of integers is provided,
        the model will be trained with each of the provided values. If a range object is provided,
        the model will be trained with each of the values in the range.

        Returns:
            list[int] | range: The number of nearest neighbours to consider.
        """

        # Make copy if list
        if isinstance(self._knn_n_neighbours, list):
            return self._knn_n_neighbours.copy()

        return self._knn_n_neighbours

    @property
    def knn_weights(self) -> list[Literal["uniform", "distance"]]:
        """
        The weights to use for the K-Nearest Neighbors algorithm.

        The possible values are 'uniform' and 'distance'. 'uniform' will assign equal weights to all
        neighbours, while 'distance' will assign weights to the neighbours based on their distance.

        Returns:
            list[Literal["uniform", "distance"]]: The weights to use for the K-Nearest Neighbors algorithm.
        """
        return self._knn_weights.copy()

    @property
    def knn_metrics(self) -> list[str]:
        """
        The metrics to use for the K-Nearest Neighbors algorithm.

        The possible values are the metrics available in the scikit-learn library, such as 'minkowski',
        'cosine', 'manhattan', etc.

        Returns:
            list[str]: The metrics to use for the K-Nearest Neighbors algorithm.
        """
        return self._knn_metrics.copy()

    @property
    def pca_n_components(self) -> list[float] | range:
        """
        The number of components to retain after applying PCA.

        This can be either a list of floats between 0 and 1, or a range object. If a list of floats is provided,
        the number of components to retain will be the length of the list. If a range object is provided, the
        number of components to retain will be the length of the range.

        Returns:
            list[float] | range: The number of components to retain after applying PCA.
        """
        
        # Make copy if list
        if isinstance(self._pca_n_components, list):
            return self._pca_n_components.copy()

        return self._pca_n_components

    @folds.setter
    def folds(self, folds: int) -> None:
        """
        Sets the number of folds for cross-validation.

        Args:
            folds (int): The number of folds for cross-validation.

        Raises:
            TypeError: If the provided value is not an integer.
        """

        # Check type
        if not isinstance(folds, int):
            msg = INVALID_TYPE_ERROR.format(variable="folds", type="int")
            raise TypeError(msg)

        self._folds = folds

    @validation_metric.setter
    def validation_metric(self, validation_metric: str) -> None:
        """
        Sets the metric to use for validation.

        Args:
            validation_metric (str): The metric to use for validation.

        Raises:
            TypeError: If the provided value is not a string.
        """
        # Check type
        if not isinstance(validation_metric, str):
            msg = INVALID_TYPE_ERROR.format(variable="validation_metric", type="str")
            raise TypeError(msg)

        self._validation_metric = validation_metric

    @best_row_strategy.setter
    def best_row_strategy(self, best_row_strategy: Literal["max", "min"]) -> None:
        """
        Sets the strategy to use for selecting the best row.

        Args:
            best_row_strategy (Literal["max", "min"]): The strategy to use for selecting the best row.

        Raises:
            TypeError: If the provided value is not "max" or "min".
        """
        
        # Check type
        if best_row_strategy not in ["max", "min"]:
            msg = INVALID_TYPE_ERROR.format(
                variable="best_row_strategy", type="max/min"
            )
            raise TypeError(msg)

        self._best_row_strategy = best_row_strategy

    @knn_n_neighbours.setter
    def knn_n_neighbours(self, knn_n_neighbours: list[int] | range) -> None:
        """
        Sets the number of nearest neighbours to consider when using the K-Nearest Neighbors algorithm.

        This can be either a list of integers or a range object. If a list of integers is provided,
        the model will be trained with each of the provided values. If a range object is provided,
        the model will be trained with each of the values in the range.

        All values must be at least 1. If a list is provided, all values must be integers.

        Args:
            knn_n_neighbours (list[int] | range): The number of nearest neighbours to consider.
        Raises:
            TypeError: If the provided value is not a list or range.
            ValueError: If any of the values are less than 1.
        """

        # Check type
        if not isinstance(knn_n_neighbours, (list, range)):
            msg = INVALID_TYPE_ERROR.format(
                variable="knn_n_neighbours", type="list or range"
            )
            raise TypeError(msg)

        # Check values if list
        if isinstance(knn_n_neighbours, list) and min(knn_n_neighbours) < 1:
            msg = "knn_n_neighbours must be at least 1"
            raise ValueError(msg)

        # Check values if range
        if isinstance(knn_n_neighbours, range) and (
            knn_n_neighbours.start < 1 or knn_n_neighbours.stop < 0
        ):
            msg = "knn_n_neighbours must be at least 1"
            raise ValueError(msg)

        # Check if all values are int
        if not all(isinstance(item, int) for item in knn_n_neighbours):
            msg = INVALID_TYPE_ERROR.format(
                variable="knn_n_neighbours", type="list or range of int"
            )
            raise ValueError(msg)

        self._knn_n_neighbours = knn_n_neighbours

    @knn_weights.setter
    def knn_weights(self, knn_weights: list[Literal["uniform", "distance"]]) -> None:
        """
        Sets the weights to use for the K-Nearest Neighbors algorithm.

        The possible values are 'uniform' and 'distance'. 'uniform' will assign equal weights to all
        neighbours, while 'distance' will assign weights to the neighbours based on their distance.

        Args:
            knn_weights (list[Literal["uniform", "distance"]]): The weights to use for the K-Nearest Neighbors algorithm.
        Raises:
            TypeError: If the provided value is not a list.
            ValueError: If any of the values are not 'uniform' or 'distance'.
        """

        # Check type
        if not isinstance(knn_weights, list):
            msg = INVALID_TYPE_ERROR.format(variable="knn_weights", type="list")
            raise TypeError(msg)

        # Check values
        if set(knn_weights) != set(["uniform", "distance"]):
            msg = INVALID_TYPE_ERROR.format(
                variable="knn_weights", type='list of "uniform"/"distance"'
            )
            raise ValueError(msg)

        self._knn_weights = knn_weights

    @knn_metrics.setter
    def knn_metrics(self, knn_metrics: list[str]) -> None:
        """
        Sets the metrics to use for the K-Nearest Neighbors algorithm.

        The possible values are 'minkowski' and 'cosine'. 'minkowski' will use the Lp norm,
        while 'cosine' will use the cosine similarity.

        Args:
            knn_metrics (list[str]): The metrics to use for the K-Nearest Neighbors algorithm.

        Raises:
            TypeError: If the provided value is not a list.
            ValueError: If any of the values are not 'minkowski' or 'cosine'.
        """

        # Check type
        if not isinstance(knn_metrics, list):
            msg = INVALID_TYPE_ERROR.format(variable="knn_metrics", type="list")
            raise TypeError(msg)

        # Check values
        if not all(isinstance(item, str) for item in knn_metrics):
            msg = INVALID_TYPE_ERROR.format(variable="knn_metrics", type="list of str")
            raise TypeError(msg)

        self._knn_metrics = knn_metrics

    @pca_n_components.setter
    def pca_n_components(self, pca_n_components: list[float | int] | range) -> None:
        """
        Sets the number of components to retain after applying PCA.

        This can be either a list of floats or a range object. If a list of floats is provided,
        the number of components to retain will be the length of the list. If a range object is provided,
        the number of components to retain will be the length of the range.

        All values must be between 0 and 1. If a list is provided, all values must be floats.

        Args:
            pca_n_components (list[float] | range): The number of components to retain after applying PCA.

        Raises:
            TypeError: If the provided value is not a list or range.
            ValueError: If any of the values are not between 0 and 1.
        """

        # Check type
        if not isinstance(pca_n_components, (list, range)):
            msg = INVALID_TYPE_ERROR.format(
                variable="knn_n_neighbours", type="list or range"
            )
            raise TypeError(msg)

        # Check values
        for item in pca_n_components:
            # Convert float to int
            if float(item) == int(item):
                item = int(item)

            # Check if between 0 and 1 if float
            if isinstance(item, float) and not (0 < item < 1):
                msg = "pca_n_components must be between 0 and 1 if pca_n_components is a float"
                raise ValueError(msg)

            # Check if at least 1 if int
            if isinstance(item, int) and item < 1:
                msg = "pca_n_components must be at least 1 if pca_n_components is an integer"
                raise ValueError(msg)

        self._pca_n_components = pca_n_components
